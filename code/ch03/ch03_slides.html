NUMA Topology Binding

#!/bin/bash
# NUMA topology binding script
# Binds only when the platform exposes an authoritative GPU NUMA node.

mapfile -t GPUS < <(nvidia-smi --query-gpu=index --format=csv,noheader)

for GPU in "${GPUS[@]}"; do
    PCI_BUS=$(nvidia-smi -i "$GPU" --query-gpu=pci.bus_id --format=csv,noheader | head -1 | tr -d '[:space:]' | tr '[:upper:]' '[:lower:]')
    SYSFS_BUS="$PCI_BUS"
    if [[ "$SYSFS_BUS" == 00000000:* ]]; then
        SYSFS_BUS="${SYSFS_BUS:4}"
    fi

    NODE=""
    NUMA_PATH="/sys/bus/pci/devices/${SYSFS_BUS}/numa_node"
    if [[ -r "$NUMA_PATH" ]]; then
        NODE=$(cat "$NUMA_PATH")
    fi

    if [[ "$NODE" =~ ^[0-9]+$ ]]; then
        numactl --cpunodebind="$NODE" --membind="$NODE" \
                bash -c "CUDA_VISIBLE_DEVICES=$GPU python train.py --gpu $GPU" &
    else
        echo "GPU $GPU NUMA node unavailable; running without numactl binding" >&2
        bash -c "CUDA_VISIBLE_DEVICES=$GPU python train.py --gpu $GPU" &
    fi
done

# Wait for all background processes to complete
wait

System-Wide Performance Tuning

#!/bin/bash
# System-wide performance tuning for GPU environments
# Run as root or with sudo

echo "Applying system-wide GPU performance optimizations..."

# 1. CPU Governor and C-states
echo "Setting CPU governor to performance mode..."
cpupower frequency-set -g performance

# 2. Virtual Memory and Swapping
echo "Disabling swap and setting swappiness to 0..."
swapoff -a
echo 0 > /proc/sys/vm/swappiness

# 3. Transparent Huge Pages
echo "Configuring Transparent Huge Pages..."
echo always > /sys/kernel/mm/transparent_hugepage/enabled

# 4. GPU-specific optimizations
echo "Configuring GPU settings..."
# Enable persistence mode for all GPUs
nvidia-smi -pm 1

# 5. NUMA balancing
echo "Disabling automatic NUMA balancing..."
echo 0 > /proc/sys/kernel/numa_balancing

Pinned-Memory Prefetch MLP

Baseline: Blocking Copies

def benchmark_fn(self):
    idx = self.batch_idx % len(self.host_batches)
    host_x = self.host_batches[idx]
    host_y = self.targets[idx]
    
    # Blocking copy (tensor not pinned)
    x = self.to_device(host_x)
    y = self.to_device(host_y)
    out = self.model(x)
    loss = mse_loss(out, y)
    loss.backward()
    self.optimizer.step()

Problem: Synchronous H2D copies block the GPU

Optimized: Pinned + Prefetch

class Prefetcher:
    def _prefetch(self):
        with torch.cuda.stream(self.copy_stream):
            self.buffers[self.next_slot].copy_(
                self.host_batches[host_idx], 
                non_blocking=True)
    
    def next(self):
        torch.cuda.current_stream().wait_stream(
            self.copy_stream)
        # Swap buffers and prefetch next
        self.cur_slot, self.next_slot = self.next_slot, self.cur_slot
        self._prefetch()
        return self.buffers[self.cur_slot]

Solution: Pinned memory + async prefetch + double buffering

Double-Buffered Batch Provisioning

Baseline: Sequential Processing

def benchmark_fn(self):
    # Get next batch
    batch = self.batches[self.batch_idx]
    
    # Process sequentially
    result = self.process_batch(batch)
    
    # Wait for completion
    torch.cuda.synchronize()
    
    self.batch_idx += 1
    return result

Problem: No overlap between batches

Optimized: Double Buffering

class DoubleBuffer:
    def __init__(self):
        self.buffer = [
            torch.empty(...),
            torch.empty(...)
        ]
        self.current = 0
    
    def next_batch(self):
        # Process current buffer
        result = self.process(self.buffer[self.current])
        
        # Swap and prepare next
        self.current = 1 - self.current
        self.prepare_next(self.buffer[self.current])
        
        return result

Solution: Overlap copy with compute using double buffers

Docker GPU Optimization

# Multi-stage Docker build for optimized GPU performance
FROM nvcr.io/nvidia/pytorch:25.11-py3 AS base

# Install additional dependencies for performance optimization
RUN apt-get update && apt-get install -y \
    numactl \
    libnuma-dev \
    htop \
    iotop \
    tcmalloc-minimal \
    jemalloc \
    && rm -rf /var/lib/apt/lists/*

# Set environment variables for optimized memory allocation
ENV MALLOC_CONF="narenas:8,dirty_decay_ms:10000,muzzy_decay_ms:10000,background_thread:true"
ENV TCMALLOC_MAX_TOTAL_THREAD_CACHE_BYTES=536870912
ENV TCMALLOC_RELEASE_RATE=16

# Set CUDA environment variables
ENV PYTORCH_ALLOC_CONF=max_split_size_mb:512
ENV TORCH_CUDA_ARCH_LIST="10.0;10.3;12.0;12.1+PTX"
ENV CUDA_DEVICE_ORDER=PCI_BUS_ID

# Runtime stage for smaller final image
FROM base AS runtime

# Set up runtime environment
ENV PYTHONPATH=/app
ENV OMP_NUM_THREADS=1
ENV MKL_NUM_THREADS=1

# Entry point with NUMA awareness
ENTRYPOINT ["numactl", "--interleave=all", "python"]
CMD ["train.py"]

Key Takeaways

1. Host is the Bottleneck

Before optimizing kernels, ensure host configuration isn't limiting GPU throughput. NUMA, memory, and CPU settings matter.

2. Pinned Memory is Critical

For async H2D transfers, pinned memory enables non-blocking copies that overlap with compute.

3. Double Buffering Overlaps

Use double buffers to overlap data transfer with computation, hiding latency.

4. Container Matters

Docker and Kubernetes settings can silently limit GPU performance. Configure memory allocators and NUMA awareness.
