Real-time GPU Streaming

Server-Sent Events for live GPU telemetry

# Backend: SSE event stream (server.py) @fastapi_app.get("/api/gpu/stream") async def gpu_stream(request, interval=5.0): async def _event_stream(): while True: if await request.is_disconnected(): break payload = { "timestamp": time.strftime(...), "gpu": get_engine().gpu.info(), } yield f"event: gpu\ndata: {json.dumps(payload)}\n\n" await asyncio.sleep(interval) return StreamingResponse( _event_stream(), media_type="text/event-stream" )
// Frontend: useGpuStream hook export function useGpuStream({ enabled, intervalMs }) { const [gpu, setGpu] = useState(null); useEffect(() => { const es = new EventSource(`/api/gpu/stream?interval=${intervalMs/1000}`); es.addEventListener("gpu", (e) => { const data = JSON.parse(e.data); setGpu(data.gpu); }); return () => es.close(); }, [enabled, intervalMs]); return { gpu, lastUpdated }; }
